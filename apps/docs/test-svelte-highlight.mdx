## Svelte Example
```svelte
<script>
  export let name = 'world';
  let count = 0;

  function handleClick() {
    count += 1;
  }
</script>

<style>
  h1 {
    color: orange;
  }
</style>

<h1>Hello {name}!</h1>
<button on:click={handleClick}>
  Clicked {count} {count === 1 ? 'time' : 'times'}
</button>
```

## React Example
```jsx
import React, { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <h1>Count: {count}</h1>
      <button onClick={() => setCount(count + 1)}>
        Increment
      </button>
    </div>
  );
}
```
